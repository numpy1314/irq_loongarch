1// Ref: Linux irq-loongson-pch-pic.c
2use crate::mmio;
3
4const PIC_COUNT_PER_REG: usize = 32;
5const PIC_REG_COUNT: usize = 2;
6
7const PCH_PIC_MASK: usize = 0x20;
8const PCH_PIC_EDGE: usize = 0x60;
9const PCH_PIC_POL: usize = 0x3e0;
10const PCH_INT_HTVEC: usize = 0x200;
11
12#[derive(Copy, Clone)]
13pub struct PchPic {
14    mmio_base: usize, // 这里传入“已经映射到虚拟地址空间的 base”
15}
16
17impl PchPic {
18    /// # Safety
19    /// 调用者必须保证 mmio_base 指向正确的 PCH-PIC MMIO 映射区域，并具备访问权限。
20    pub const unsafe fn new(mmio_base: usize) -> Self {
21        Self { mmio_base }
22    }
23
24    #[inline(always)]
25    fn read_w(&self, off: usize) -> u32 {
26        unsafe { mmio::read32(self.mmio_base + off) }
27    }
28
29    #[inline(always)]
30    fn write_w(&self, off: usize, val: u32) {
31        unsafe { mmio::write32(self.mmio_base + off, val) }
32    }
33
34    pub fn init(&self) {
35        // High level triggered
36        for _ in 0..PIC_REG_COUNT {
37            self.write_w(PCH_PIC_EDGE, 0);
38            self.write_w(PCH_PIC_POL, 0);
39        }
40    }
41
42    fn split_bit(irq: usize) -> (usize, u32) {
43        (irq / PIC_COUNT_PER_REG * 4, 1 << (irq % PIC_COUNT_PER_REG))
44    }
45
46    pub fn enable_irq(&self, irq: usize) {
47        let (offset, bit) = Self::split_bit(irq);
48
49        let addr = PCH_PIC_MASK + offset;
50        self.write_w(addr, self.read_w(addr) & !bit);
51
52        // route vector
53        let addr = PCH_INT_HTVEC + irq;
54        unsafe { mmio::write8(self.mmio_base + addr, irq as u8) };
55    }
56
57    pub fn disable_irq(&self, irq: usize) {
58        let (offset, bit) = Self::split_bit(irq);
59        let addr = PCH_PIC_MASK + offset;
60        self.write_w(addr, self.read_w(addr) | bit);
61    }
62}
